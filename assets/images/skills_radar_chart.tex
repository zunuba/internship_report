% Skills Development Radar Chart
% TikZ diagram for Chapter 3

\documentclass[tikz,border=10pt]{standalone}
\usepackage{tikz}
\usetikzlibrary{shapes,arrows,positioning,fit,backgrounds}

\begin{document}

\begin{tikzpicture}[
    scale=1.2,
    every node/.style={font=\small}
]

% Title
\node[text width=12cm, text centered, font=\Large\bfseries] (title) at (0,8) {Skills Development Assessment - Before vs After Internship};

% Define skills and their positions
\def\skills{{
    "Financial\\Reporting",
    "Document\\Management", 
    "Budget\\Control",
    "Excel\\Proficiency",
    "Communication",
    "Problem\\Solving",
    "Team\\Collaboration",
    "Industry\\Knowledge"
}}

% Define skill levels (0-5 scale)
\def\beforelevels{1,2,1,3,2,2,2,1}
\def\afterlevels{4,4,3,4,4,4,4,3}

% Draw radar grid
\foreach \i in {1,2,3,4,5} {
    \draw[gray!30, thick] (0,0) circle (\i*0.8);
    \node[gray!50, font=\tiny] at (\i*0.8, 0.1) {\i};
}

% Draw skill axes
\foreach \i in {0,1,2,3,4,5,6,7} {
    \pgfmathsetmacro{\angle}{45*\i}
    \draw[gray!40, thick] (0,0) -- (\angle:4.2);
    \pgfmathsetmacro{\x}{\angle:4.5}
    \pgfmathsetmacro{\y}{\angle:4.5}
    \node[font=\tiny, text width=1.5cm, text centered] at (\x, \y) {
        \pgfmathparse{\skills[\i]}
        \pgfmathresult
    };
}

% Draw before internship skills (dashed line)
\draw[blue!60, thick, dashed] 
    plot[smooth, tension=0.3] coordinates {
        (0:0.8) (45:1.6) (90:0.8) (135:1.6) (180:2.4) (225:1.6) (270:1.6) (315:0.8) (360:0.8)
    };

% Draw after internship skills (solid line)
\draw[green!70, thick] 
    plot[smooth, tension=0.3] coordinates {
        (0:3.2) (45:3.2) (90:2.4) (135:3.2) (180:3.2) (225:3.2) (270:3.2) (315:2.4) (360:3.2)
    };

% Add data points for before
\foreach \i in {0,1,2,3,4,5,6,7} {
    \pgfmathsetmacro{\angle}{45*\i}
    \pgfmathsetmacro{\level}{\beforelevels[\i]}
    \pgfmathsetmacro{\radius}{\level*0.8}
    \fill[blue!60] (\angle:\radius) circle (2pt);
}

% Add data points for after
\foreach \i in {0,1,2,3,4,5,6,7} {
    \pgfmathsetmacro{\angle}{45*\i}
    \pgfmathsetmacro{\level}{\afterlevels[\i]}
    \pgfmathsetmacro{\radius}{\level*0.8}
    \fill[green!70] (\angle:\radius) circle (2pt);
}

% Legend
\node[text width=3cm, text centered, font=\small\bfseries] (legend_title) at (6,6) {Legend:};
\node[text width=3cm, text centered, font=\tiny] (legend_before) at (6,5.5) {Before Internship\\\textcolor{blue!60}{\rule{1cm}{0.1cm}}};
\node[text width=3cm, text centered, font=\tiny] (legend_after) at (6,5) {After Internship\\\textcolor{green!70}{\rule{1cm}{0.1cm}}};

% Skill level scale
\node[text width=3cm, text centered, font=\small\bfseries] (scale_title) at (6,4) {Skill Level Scale:};
\node[text width=3cm, text centered, font=\tiny] (scale1) at (6,3.5) {1 - Beginner};
\node[text width=3cm, text centered, font=\tiny] (scale2) at (6,3.2) {2 - Basic};
\node[text width=3cm, text centered, font=\tiny] (scale3) at (6,2.9) {3 - Intermediate};
\node[text width=3cm, text centered, font=\tiny] (scale4) at (6,2.6) {4 - Advanced};
\node[text width=3cm, text centered, font=\tiny] (scale5) at (6,2.3) {5 - Expert};

% Key improvements
\node[text width=4cm, text centered, font=\small\bfseries] (improvement_title) at (-6,6) {Key Improvements:};
\node[text width=4cm, text centered, font=\tiny] (improvement1) at (-6,5.5) {• Financial Reporting: 1→4};
\node[text width=4cm, text centered, font=\tiny] (improvement2) at (-6,5.2) {• Document Management: 2→4};
\node[text width=4cm, text centered, font=\tiny] (improvement3) at (-6,4.9) {• Communication: 2→4};
\node[text width=4cm, text centered, font=\tiny] (improvement4) at (-6,4.6) {• Problem Solving: 2→4};
\node[text width=4cm, text centered, font=\tiny] (improvement5) at (-6,4.3) {• Team Collaboration: 2→4};

% Overall assessment
\node[text width=4cm, text centered, font=\small\bfseries] (assessment_title) at (-6,3.5) {Overall Assessment:};
\node[text width=4cm, text centered, font=\tiny] (assessment) at (-6,3) {Average improvement:\\2.1 → 3.4\\(+62\% increase)};

\end{tikzpicture}

\end{document}
